##### 下面代码的输出结果

```js
const arr = [10, 12, 15, 21];
for (var i = 0; i < arr.length; i++) {
  setTimeout(function () {
    console.log("The index of this number is: " + i);
  }, 3000);
}
```

##### 如何改造上面的代码输出结果

```js
const arr = [10, 12, 15, 21];
for (var i = 0; i < arr.length; i++) {
  setTimeout(function (i) {
      return function () {
        console.log("The index of this number is: " + i);
      };
    }(i),  3000
  );
}
```

```js
const arr = [10, 12, 15, 21];
for (let i = 0; i < arr.length; i++) {
```
